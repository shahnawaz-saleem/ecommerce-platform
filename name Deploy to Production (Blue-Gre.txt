name: Deploy to Production (Blue-Green)

on:
push:
branches:
- main

jobs:
build-and-deploy:
runs-on: ubuntu-latest

```
steps:
  # 1. Checkout code
  - name: Checkout
    uses: actions/checkout@v4

  # 2. Setup .NET
  - name: Setup .NET
    uses: actions/setup-dotnet@v4
    with:
      dotnet-version: '8.0.x'

  # 3. Restore
  - name: Restore dependencies
    run: dotnet restore

  # 4. Build
  - name: Build
    run: dotnet build --no-restore --configuration Release

  # 5. Publish
  - name: Publish
    run: dotnet publish -c Release -o ./publish

  # 6. Deploy to STAGING slot (Green)
  - name: Deploy to Azure Web App (Staging Slot)
    uses: azure/webapps-deploy@v2
    with:
      app-name: 'catalog-service-prod'
      slot-name: 'staging'
      publish-profile: ${{ secrets.AZURE_PUBLISH_PROFILE }}
      package: ./publish

  # 7. Wait for app to warm up
  - name: Wait for warmup
    run: sleep 30

  # 8. Health check
  - name: Health Check
    run: |
      curl --fail https://catalog-service-prod-staging.azurewebsites.net/health

  # 9. Swap slots (Blue-Green)
  - name: Swap Slots
    uses: azure/cli@v2
    with:
      inlineScript: |
        az webapp deployment slot swap \
          --name catalog-service-prod \
          --resource-group YOUR_RESOURCE_GROUP \
          --slot staging \
          --target-slot production
```
